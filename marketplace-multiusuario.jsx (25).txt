import React, { useState, useRef, useEffect } from 'react';
import { LogOut, Plus, ShoppingBag, MessageSquare, User, Home, BarChart3, Bell, X, Send, Eye, EyeOff, Search } from 'lucide-react';

const MarketplaceMultiusuario = () => {
  // ============ ESTADO GLOBAL ============
  const [usuarioLogado, setUsuarioLogado] = useState(null);
  const [telaAtual, setTelaAtual] = useState('login');
  const [usuarios, setUsuarios] = useState({
    'vendedor@gmail.com': { id: 'v1', email: 'vendedor@gmail.com', nome: 'João Vendedor', senha: '123456', tipo: 'vendedor' },
    'comprador@gmail.com': { id: 'c1', email: 'comprador@gmail.com', nome: 'Maria Compradora', senha: '123456', tipo: 'comprador' }
  });

  const [produtos, setProdutos] = useState([
    {
      id: 'p1',
      vendedorId: 'v1',
      nome: 'iPhone 13',
      categoria: 'Eletrônicos',
      preco: 2500,
      precoMinimo: 2000,
      imagem: '📱',
      descricao: 'iPhone 13, 128GB, rosa, excelente estado',
      status: 'ativo',
      dataCriacao: new Date(),
      negociacoes: []
    },
    {
      id: 'p2',
      vendedorId: 'v1',
      nome: 'MacBook Air M1',
      categoria: 'Eletrônicos',
      preco: 5000,
      precoMinimo: 4000,
      imagem: '💻',
      descricao: 'MacBook Air M1, 256GB SSD, como novo',
      status: 'ativo',
      dataCriacao: new Date(),
      negociacoes: []
    }
  ]);

  const [negociacoes, setNegociacoes] = useState([]);
  const [notificacoes, setNotificacoes] = useState([]);
  
  // Estados de UI
  const [showNovoSenha, setShowNovoSenha] = useState(false);
  const [emailLogin, setEmailLogin] = useState('');
  const [senhaLogin, setSenhaLogin] = useState('');
  const [emailCadastro, setEmailCadastro] = useState('');
  const [nomeCadastro, setNomeCadastro] = useState('');
  const [senhaCadastro, setSenhaCadastro] = useState('');
  const [tipoCadastro, setTipoCadastro] = useState('comprador');
  
  const [novoProduto, setNovoProduto] = useState({ nome: '', categoria: 'Eletrônicos', preco: '', precoMinimo: '', descricao: '', imagem: '📦' });
  const [produtoEmChat, setProdutoEmChat] = useState(null);
  const [mensagensChat, setMensagensChat] = useState([]);
  const [inputChat, setInputChat] = useState('');
  const [filtroCategoria, setFiltroCategoria] = useState('Todos');
  const [busca, setBusca] = useState('');
  const [ofertaAtual, setOfertaAtual] = useState('');
  
  const messagesEndRef = useRef(null);
  const scrollToBottom = () => messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  useEffect(() => scrollToBottom(), [mensagensChat]);

  const categorias = ['Eletrônicos', 'Moda', 'Casa', 'Esportes', 'Livros', 'Carros'];

  // ============ AUTENTICAÇÃO ============
  const handleLogin = () => {
    const usuario = usuarios[emailLogin];
    if (usuario && usuario.senha === senhaLogin) {
      setUsuarioLogado(usuario);
      setTelaAtual(usuario.tipo === 'vendedor' ? 'dashboard-vendedor' : 'explorar');
      setEmailLogin('');
      setSenhaLogin('');
    } else {
      alert('Email ou senha inválidos!');
    }
  };

  const handleCadastro = () => {
    if (!emailCadastro || !nomeCadastro || !senhaCadastro) {
      alert('Preencha todos os campos!');
      return;
    }
    if (usuarios[emailCadastro]) {
      alert('Email já cadastrado!');
      return;
    }
    const novoUsuario = {
      id: Date.now().toString(),
      email: emailCadastro,
      nome: nomeCadastro,
      senha: senhaCadastro,
      tipo: tipoCadastro
    };
    setUsuarios(prev => ({ ...prev, [emailCadastro]: novoUsuario }));
    setUsuarioLogado(novoUsuario);
    setTelaAtual(tipoCadastro === 'vendedor' ? 'dashboard-vendedor' : 'explorar');
    setEmailCadastro('');
    setNomeCadastro('');
    setSenhaCadastro('');
  };

  const handleLogout = () => {
    setUsuarioLogado(null);
    setTelaAtual('login');
    setEmailLogin('');
    setSenhaLogin('');
  };

  // ============ PRODUTOS ============
  const adicionarProduto = () => {
    if (!novoProduto.nome || !novoProduto.preco || !novoProduto.precoMinimo) {
      alert('Preencha todos os campos!');
      return;
    }
    const produto = {
      id: Date.now().toString(),
      vendedorId: usuarioLogado.id,
      nome: novoProduto.nome,
      categoria: novoProduto.categoria,
      preco: parseInt(novoProduto.preco),
      precoMinimo: parseInt(novoProduto.precoMinimo),
      imagem: novoProduto.imagem,
      descricao: novoProduto.descricao || 'Produto em bom estado',
      status: 'ativo',
      dataCriacao: new Date(),
      negociacoes: []
    };
    setProdutos(prev => [...prev, produto]);
    setNovoProduto({ nome: '', categoria: 'Eletrônicos', preco: '', precoMinimo: '', descricao: '', imagem: '📦' });
    alert('Produto adicionado com sucesso!');
  };

  const deletarProduto = (id) => {
    setProdutos(prev => prev.filter(p => p.id !== id));
  };

  // ============ NEGOCIAÇÃO ============
  const abrirChat = (produto) => {
    setProdutoEmChat(produto);
    setMensagensChat([
      {
        id: 1,
        tipo: 'bot',
        texto: `👋 Olá! Tenho interesse em "${produto.nome}". Qual é o menor preço que você aceita?`,
        timestamp: new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' })
      }
    ]);
    setOfertaAtual(produto.preco.toString());
  };

  const fecharChat = () => {
    setProdutoEmChat(null);
    setMensagensChat([]);
    setInputChat('');
    setOfertaAtual('');
  };

  const gerarRespostaNegociacao = (mensagem, produto, ofertaComprador) => {
    const msgLower = mensagem.toLowerCase();
    const precoAtual = parseInt(ofertaComprador) || produto.preco;
    const desconto = ((produto.preco - precoAtual) / produto.preco * 100).toFixed(0);

    const respostas = {
      caro: [
        `💭 Entendo, é um investimento. Você pode chegar a R$ ${(precoAtual + 200).toLocaleString('pt-BR')}?`,
        `🤔 Consigo fazer uma pequena redução. Que tal R$ ${(precoAtual + 150).toLocaleString('pt-BR')}?`
      ],
      desconto: [
        `✅ ${desconto}% de desconto é bom! Vamos de R$ ${precoAtual.toLocaleString('pt-BR')}?`,
        `💰 Posso fazer R$ ${Math.max(precoAtual + 100, produto.precoMinimo).toLocaleString('pt-BR')} se vir buscar hoje.`
      ],
      minimo: [
        `⚠️ Esse é meu preço mínimo: R$ ${produto.precoMinimo.toLocaleString('pt-BR')}. Fechamos?`,
        `💯 Consigo chegar a R$ ${produto.precoMinimo.toLocaleString('pt-BR')} no máximo. Aceita?`
      ],
      visto: [
        `👀 Viu bem? É excelente custo-benefício!`,
        `✨ Produto novo, sem defeitos. Você aprova?`
      ],
      compra: [
        `🎉 Perfeito! Vamos finalizar? Como você prefere - busca ou entrega?`,
        `🙌 Ótimo! Vou preparar tudo. Qual melhor horário para você?`
      ]
    };

    if (msgLower.includes('caro') || msgLower.includes('muito') || msgLower.includes('reduz')) {
      return respostas.caro[Math.floor(Math.random() * respostas.caro.length)];
    }
    if (msgLower.includes('desconto') || msgLower.includes('menor') || parseInt(msgLower) < produto.preco) {
      return respostas.desconto[Math.floor(Math.random() * respostas.desconto.length)];
    }
    if (precoAtual <= produto.precoMinimo) {
      return respostas.minimo[Math.floor(Math.random() * respostas.minimo.length)];
    }
    if (msgLower.includes('compro') || msgLower.includes('fechar') || msgLower.includes('topei') || msgLower.includes('topa')) {
      return respostas.compra[Math.floor(Math.random() * respostas.compra.length)];
    }
    if (msgLower.includes('visto') || msgLower.includes('vê') || msgLower.includes('tem')) {
      return respostas.visto[Math.floor(Math.random() * respostas.visto.length)];
    }

    return `Entendi! Qual é sua melhor proposta? Estou aberto a negociar!`;
  };

  const enviarMensagem = () => {
    if (!inputChat.trim()) return;

    // Mensagem do comprador
    const msgComprador = {
      id: Date.now(),
      tipo: 'comprador',
      texto: inputChat,
      timestamp: new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' })
    };
    setMensagensChat(prev => [...prev, msgComprador]);

    // Atualizar oferta se for número
    if (!isNaN(inputChat)) {
      setOfertaAtual(inputChat);
    }

    // Resposta do vendedor (bot)
    const respostaTexto = gerarRespostaNegociacao(inputChat, produtoEmChat, ofertaAtual);
    const msgVendedor = {
      id: Date.now() + 1,
      tipo: 'vendedor',
      texto: respostaTexto,
      timestamp: new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' })
    };

    setTimeout(() => {
      setMensagensChat(prev => [...prev, msgVendedor]);

      // Se fechar venda
      if (inputChat.toLowerCase().includes('compro') || inputChat.toLowerCase().includes('topa')) {
        setTimeout(() => {
          const precoFinal = parseInt(ofertaAtual) || produtoEmChat.preco;
          setNotificacoes(prev => [...prev, {
            id: Date.now(),
            tipo: 'venda',
            mensagem: `✅ Venda concluída! ${produtoEmChat.nome} por R$ ${precoFinal.toLocaleString('pt-BR')}`,
            comprador: usuarioLogado.nome,
            vendedor: usuarios[Object.keys(usuarios).find(k => usuarios[k].id === produtoEmChat.vendedorId)].nome
          }]);
          setProdutos(prev => prev.map(p => p.id === produtoEmChat.id ? { ...p, status: 'vendido' } : p));
          alert(`🎉 Venda realizada por R$ ${precoFinal.toLocaleString('pt-BR')}!`);
          fecharChat();
        }, 800);
      }
    }, 1000);

    setInputChat('');
  };

  // ============ FILTROS ============
  const produtosFiltrados = usuarioLogado ? produtos.filter(p => {
    const temCategoria = filtroCategoria === 'Todos' || p.categoria === filtroCategoria;
    const temBusca = p.nome.toLowerCase().includes(busca.toLowerCase());
    const naoEhVendedor = usuarioLogado.tipo !== 'vendedor' || p.vendedorId !== usuarioLogado.id;
    return temCategoria && temBusca && naoEhVendedor && p.status === 'ativo';
  }) : [];

  const meusProductosVendedor = usuarioLogado ? produtos.filter(p => p.vendedorId === usuarioLogado.id) : [];

  // ============ TELAS ============

  // Login/Cadastro
  if (telaAtual === 'login' || telaAtual === 'cadastro') {
    return (
      <div className="min-h-screen bg-gradient-to-br from-blue-600 to-purple-700 flex items-center justify-center p-4">
        <div className="w-full max-w-md">
          <div className="bg-white rounded-2xl shadow-2xl p-8">
            <h1 className="text-3xl font-bold text-gray-900 mb-2 text-center">🏪 MarketHub</h1>
            <p className="text-gray-600 text-center mb-8">Compre e venda com negociação automática</p>

            {telaAtual === 'login' ? (
              <div className="space-y-4">
                <input
                  type="email"
                  placeholder="Email"
                  value={emailLogin}
                  onChange={(e) => setEmailLogin(e.target.value)}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                />
                <div className="relative">
                  <input
                    type={showNovoSenha ? 'text' : 'password'}
                    placeholder="Senha"
                    value={senhaLogin}
                    onChange={(e) => setSenhaLogin(e.target.value)}
                    onKeyPress={(e) => e.key === 'Enter' && handleLogin()}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                  />
                  <button
                    onClick={() => setShowNovoSenha(!showNovoSenha)}
                    className="absolute right-3 top-3 text-gray-500"
                  >
                    {showNovoSenha ? <EyeOff size={20} /> : <Eye size={20} />}
                  </button>
                </div>
                <button
                  onClick={handleLogin}
                  className="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 rounded-lg transition"
                >
                  Entrar
                </button>
                <p className="text-center text-gray-600 text-sm">
                  Não tem conta?{' '}
                  <button
                    onClick={() => setTelaAtual('cadastro')}
                    className="text-blue-600 hover:underline font-semibold"
                  >
                    Cadastre-se
                  </button>
                </p>
                <div className="pt-4 border-t border-gray-300 text-center text-xs text-gray-500">
                  <p className="mb-2">Contas de teste:</p>
                  <p>Vendedor: vendedor@gmail.com / 123456</p>
                  <p>Comprador: comprador@gmail.com / 123456</p>
                </div>
              </div>
            ) : (
              <div className="space-y-4">
                <input
                  type="text"
                  placeholder="Nome completo"
                  value={nomeCadastro}
                  onChange={(e) => setNomeCadastro(e.target.value)}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                />
                <input
                  type="email"
                  placeholder="Email"
                  value={emailCadastro}
                  onChange={(e) => setEmailCadastro(e.target.value)}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                />
                <div className="relative">
                  <input
                    type={showNovoSenha ? 'text' : 'password'}
                    placeholder="Senha"
                    value={senhaCadastro}
                    onChange={(e) => setSenhaCadastro(e.target.value)}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                  />
                  <button
                    onClick={() => setShowNovoSenha(!showNovoSenha)}
                    className="absolute right-3 top-3 text-gray-500"
                  >
                    {showNovoSenha ? <EyeOff size={20} /> : <Eye size={20} />}
                  </button>
                </div>
                <select
                  value={tipoCadastro}
                  onChange={(e) => setTipoCadastro(e.target.value)}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                >
                  <option value="comprador">Sou Comprador</option>
                  <option value="vendedor">Sou Vendedor</option>
                </select>
                <button
                  onClick={handleCadastro}
                  className="w-full bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg transition"
                >
                  Cadastrar
                </button>
                <p className="text-center text-gray-600 text-sm">
                  Já tem conta?{' '}
                  <button
                    onClick={() => setTelaAtual('login')}
                    className="text-blue-600 hover:underline font-semibold"
                  >
                    Faça login
                  </button>
                </p>
              </div>
            )}
          </div>
        </div>
      </div>
    );
  }

  // Navbar
  const navbar = (
    <nav className="bg-gradient-to-r from-blue-600 to-purple-700 text-white shadow-lg">
      <div className="max-w-7xl mx-auto px-4 py-4 flex items-center justify-between">
        <div className="flex items-center gap-2">
          <ShoppingBag size={28} />
          <h1 className="text-2xl font-bold">🏪 MarketHub</h1>
        </div>
        <div className="flex items-center gap-4">
          <span className="text-sm">👤 {usuarioLogado.nome}</span>
          <button
            onClick={() => setNotificacoes([])}
            className="relative hover:bg-white/20 p-2 rounded-lg transition"
          >
            <Bell size={24} />
            {notificacoes.length > 0 && (
              <span className="absolute top-0 right-0 bg-red-500 text-white text-xs px-2 py-1 rounded-full">
                {notificacoes.length}
              </span>
            )}
          </button>
          <button
            onClick={handleLogout}
            className="flex items-center gap-2 hover:bg-white/20 px-4 py-2 rounded-lg transition"
          >
            <LogOut size={20} /> Sair
          </button>
        </div>
      </div>
    </nav>
  );

  // Dashboard Vendedor
  if (telaAtual === 'dashboard-vendedor') {
    return (
      <div className="min-h-screen bg-gray-100">
        {navbar}
        <div className="max-w-7xl mx-auto p-6">
          <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mb-6">
            <button
              onClick={() => setTelaAtual('novo-produto')}
              className="bg-gradient-to-r from-green-500 to-emerald-500 hover:from-green-600 hover:to-emerald-600 text-white font-bold py-4 rounded-lg flex items-center justify-center gap-3 transition shadow-lg"
            >
              <Plus size={24} /> Novo Produto
            </button>
            <button
              onClick={() => setTelaAtual('explorar')}
              className="bg-gradient-to-r from-blue-500 to-cyan-500 hover:from-blue-600 hover:to-cyan-600 text-white font-bold py-4 rounded-lg flex items-center justify-center gap-3 transition shadow-lg"
            >
              <Search size={24} /> Explorar Marketplace
            </button>
          </div>

          <div className="bg-white rounded-lg shadow-lg p-6">
            <h2 className="text-2xl font-bold text-gray-900 mb-6">📦 Seus Produtos</h2>
            {meusProductosVendedor.length === 0 ? (
              <p className="text-gray-500 text-center py-8">Você ainda não adicionou nenhum produto.</p>
            ) : (
              <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
                {meusProductosVendedor.map(p => (
                  <div key={p.id} className="border border-gray-200 rounded-lg overflow-hidden hover:shadow-lg transition">
                    <div className="bg-gray-100 h-32 flex items-center justify-center text-5xl">{p.imagem}</div>
                    <div className="p-4">
                      <h3 className="font-bold text-lg text-gray-900 mb-2">{p.nome}</h3>
                      <p className="text-gray-600 text-sm mb-3">{p.descricao}</p>
                      <div className="mb-3">
                        <p className="text-xl font-bold text-blue-600">R$ {p.preco.toLocaleString('pt-BR')}</p>
                        <p className="text-xs text-gray-500">Mínimo: R$ {p.precoMinimo.toLocaleString('pt-BR')}</p>
                      </div>
                      <span className={`text-xs px-2 py-1 rounded-full ${p.status === 'ativo' ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'}`}>
                        {p.status === 'ativo' ? '✅ Ativo' : '❌ Vendido'}
                      </span>
                      <button
                        onClick={() => deletarProduto(p.id)}
                        className="mt-3 w-full bg-red-500 hover:bg-red-600 text-white py-2 rounded-lg transition"
                      >
                        Remover
                      </button>
                    </div>
                  </div>
                ))}
              </div>
            )}
          </div>
        </div>
      </div>
    );
  }

  // Novo Produto
  if (telaAtual === 'novo-produto') {
    return (
      <div className="min-h-screen bg-gray-100">
        {navbar}
        <div className="max-w-2xl mx-auto p-6">
          <div className="bg-white rounded-lg shadow-lg p-8">
            <h2 className="text-2xl font-bold text-gray-900 mb-6">📦 Adicionar Novo Produto</h2>
            <div className="space-y-6">
              <div>
                <label className="block text-sm font-bold text-gray-700 mb-2">Nome do Produto</label>
                <input
                  type="text"
                  value={novoProduto.nome}
                  onChange={(e) => setNovoProduto({ ...novoProduto, nome: e.target.value })}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                />
              </div>
              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-sm font-bold text-gray-700 mb-2">Categoria</label>
                  <select
                    value={novoProduto.categoria}
                    onChange={(e) => setNovoProduto({ ...novoProduto, categoria: e.target.value })}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                  >
                    {categorias.map(c => (
                      <option key={c} value={c}>{c}</option>
                    ))}
                  </select>
                </div>
                <div>
                  <label className="block text-sm font-bold text-gray-700 mb-2">Emoji</label>
                  <input
                    type="text"
                    value={novoProduto.imagem}
                    onChange={(e) => setNovoProduto({ ...novoProduto, imagem: e.target.value })}
                    maxLength="2"
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 text-center text-2xl"
                  />
                </div>
              </div>
              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-sm font-bold text-gray-700 mb-2">Preço (R$)</label>
                  <input
                    type="number"
                    value={novoProduto.preco}
                    onChange={(e) => setNovoProduto({ ...novoProduto, preco: e.target.value })}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                  />
                </div>
                <div>
                  <label className="block text-sm font-bold text-gray-700 mb-2">Preço Mínimo (R$)</label>
                  <input
                    type="number"
                    value={novoProduto.precoMinimo}
                    onChange={(e) => setNovoProduto({ ...novoProduto, precoMinimo: e.target.value })}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                  />
                </div>
              </div>
              <div>
                <label className="block text-sm font-bold text-gray-700 mb-2">Descrição</label>
                <textarea
                  value={novoProduto.descricao}
                  onChange={(e) => setNovoProduto({ ...novoProduto, descricao: e.target.value })}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 h-24"
                />
              </div>
              <div className="flex gap-4">
                <button
                  onClick={adicionarProduto}
                  className="flex-1 bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg transition"
                >
                  Publicar Produto
                </button>
                <button
                  onClick={() => setTelaAtual('dashboard-vendedor')}
                  className="flex-1 bg-gray-400 hover:bg-gray-500 text-white font-bold py-3 rounded-lg transition"
                >
                  Cancelar
                </button>
              </div>
            </div>
          </div>
        </div>
      </div>
    );
  }

  // Explorar Marketplace
  if (telaAtual === 'explorar') {
    return (
      <div className="min-h-screen bg-gray-100">
        {navbar}
        <div className="max-w-7xl mx-auto p-6">
          <div className="bg-white rounded-lg shadow-lg p-6 mb-6">
            <div className="flex flex-col md:flex-row gap-4 mb-4">
              <input
                type="text"
                placeholder="🔍 Buscar produtos..."
                value={busca}
                onChange={(e) => setBusca(e.target.value)}
                className="flex-1 px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
              />
            </div>
            <div className="flex gap-2 flex-wrap">
              {['Todos', ...categorias].map(c => (
                <button
                  key={c}
                  onClick={() => setFiltroCategoria(c)}
                  className={`px-4 py-2 rounded-full transition font-semibold ${
                    filtroCategoria === c
                      ? 'bg-blue-600 text-white'
                      : 'bg-gray-200 text-gray-700 hover:bg-gray-300'
                  }`}
                >
                  {c}
                </button>
              ))}
            </div>
          </div>

          {produtosFiltrados.length === 0 ? (
            <div className="text-center py-12">
              <p className="text-2xl text-gray-500">😔 Nenhum produto encontrado</p>
            </div>
          ) : (
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
              {produtosFiltrados.map(produto => (
                <div key={produto.id} className="bg-white rounded-lg shadow-lg overflow-hidden hover:shadow-xl transition">
                  <div className="bg-gradient-to-br from-blue-400 to-purple-500 h-40 flex items-center justify-center text-6xl">
                    {produto.imagem}
                  </div>
                  <div className="p-6">
                    <h3 className="font-bold text-lg text-gray-900 mb-2">{produto.nome}</h3>
                    <p className="text-gray-600 text-sm mb-4">{produto.descricao}</p>
                    <div className="mb-4">
                      <p className="text-2xl font-bold text-blue-600">R$ {produto.preco.toLocaleString('pt-BR')}</p>
                      <p className="text-xs text-gray-500">Negocie até R$ {produto.precoMinimo.toLocaleString('pt-BR')}</p>
                    </div>
                    <button
                      onClick={() => abrirChat(produto)}
                      className="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 rounded-lg flex items-center justify-center gap-2 transition"
                    >
                      <MessageSquare size={20} /> Negociar
                    </button>
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    );
  }

  // Chat de Negociação
  if (produtoEmChat) {
    return (
      <div className="fixed inset-0 bg-black/50 backdrop-blur-sm flex items-end md:items-center justify-end md:justify-center p-4 z-50">
        <div className="w-full md:w-96 h-screen md:h-[600px] bg-white rounded-2xl md:rounded-xl shadow-2xl flex flex-col overflow-hidden">
          <div className="bg-gradient-to-r from-blue-600 to-purple-700 text-white p-5 flex items-center justify-between">
            <div className="flex items-center gap-3">
              <span className="text-3xl">{produtoEmChat.imagem}</span>
              <div>
                <h3 className="font-bold">{produtoEmChat.nome}</h3>
                <p className="text-xs text-blue-100">R$ {(parseInt(ofertaAtual) || produtoEmChat.preco).toLocaleString('pt-BR')}</p>
              </div>
            </div>
            <button onClick={fecharChat} className="hover:bg-white/20 p-2 rounded-lg transition">
              <X size={24} />
            </button>
          </div>

          <div className="flex-1 overflow-y-auto p-4 space-y-4 bg-gray-50">
            {mensagensChat.map(msg => (
              <div key={msg.id} className={`flex ${msg.tipo === 'comprador' ? 'justify-end' : 'justify-start'}`}>
                <div
                  className={`max-w-xs px-4 py-2 rounded-lg ${
                    msg.tipo === 'comprador'
                      ? 'bg-blue-600 text-white rounded-br-none'
                      : 'bg-gray-300 text-gray-900 rounded-bl-none'
                  }`}
                >
                  <p className="text-sm">{msg.texto}</p>
                  <p className="text-xs opacity-70 mt-1">{msg.timestamp}</p>
                </div>
              </div>
            ))}
            <div ref={messagesEndRef} />
          </div>

          <div className="border-t border-gray-200 p-4 bg-white space-y-3">
            <input
              type="text"
              placeholder="Digite sua mensagem ou ofereça um preço (ex: 2000)..."
              value={inputChat}
              onChange={(e) => setInputChat(e.target.value)}
              onKeyPress={(e) => e.key === 'Enter' && enviarMensagem()}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
            />
            <button
              onClick={enviarMensagem}
              className="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 rounded-lg flex items-center justify-center gap-2 transition"
            >
              <Send size={20} /> Enviar
            </button>
          </div>
        </div>
      </div>
    );
  }

  return null;
};

export default MarketplaceMultiusuario;
