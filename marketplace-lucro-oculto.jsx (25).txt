import React, { useState, useEffect } from 'react';
import { Send, X, LogOut, Plus, LogIn } from 'lucide-react';

const MarketplaceComLucro = () => {
  const [usuarioLogado, setUsuarioLogado] = useState(JSON.parse(localStorage.getItem('usuarioLogado')));
  const [tela, setTela] = useState(usuarioLogado ? 'inicio' : 'inicio');
  const [usuarios, setUsuarios] = useState(JSON.parse(localStorage.getItem('usuarios')) || {
    'vendedor@test.com': { id: 'v1', nome: 'João Silva', email: 'vendedor@test.com', senha: '123', tipo: 'vendedor-comprador' },
    'comprador@test.com': { id: 'c1', nome: 'Maria Santos', email: 'comprador@test.com', senha: '123', tipo: 'vendedor-comprador' }
  });

  const [produtos, setProdutos] = useState(JSON.parse(localStorage.getItem('produtos')) || [
    {
      id: 1,
      nome: 'iPhone 13',
      preco_anuncio: 2500,
      preco_minimo: 2000,
      preco_repassado: 2000,
      lucro_plataforma: 500,
      emoji: '📱',
      vendedorId: 'v1',
      vendedorNome: 'João Silva',
      dataCriacao: new Date().getTime(),
      visualizacoes: 45,
      em_negociacao: 2,
      status: 'ativo'
    },
    {
      id: 2,
      nome: 'MacBook Air M1',
      preco_anuncio: 5000,
      preco_minimo: 4000,
      preco_repassado: 4000,
      lucro_plataforma: 1000,
      emoji: '💻',
      vendedorId: 'v1',
      vendedorNome: 'João Silva',
      dataCriacao: new Date().getTime() - 3600000,
      visualizacoes: 30,
      em_negociacao: 1,
      status: 'ativo'
    }
  ]);

  const [loginEmail, setLoginEmail] = useState('');
  const [loginSenha, setLoginSenha] = useState('');
  const [cadastroNome, setCadastroNome] = useState('');
  const [cadastroEmail, setCadastroEmail] = useState('');
  const [cadastroSenha, setCadastroSenha] = useState('');

  const [novoProduto, setNovoProduto] = useState({ nome: '', preco_anuncio: '', preco_minimo: '', emoji: '📦' });
  const [chatAberto, setChatAberto] = useState(null);
  const [mensagens, setMensagens] = useState([]);
  const [inputChat, setInputChat] = useState('');
  const [ofertaAtual, setOfertaAtual] = useState('');

  const [notificacoes, setNotificacoes] = useState([]);
  const [meusProdutos, setMeusProdutos] = useState([]);
  const [negociacoesAtivas, setNegociacoesAtivas] = useState({});
  const [recomendacoes, setRecomendacoes] = useState({});

  // Salvar estado
  useEffect(() => {
    localStorage.setItem('usuarioLogado', JSON.stringify(usuarioLogado));
  }, [usuarioLogado]);

  useEffect(() => {
    localStorage.setItem('usuarios', JSON.stringify(usuarios));
  }, [usuarios]);

  useEffect(() => {
    localStorage.setItem('produtos', JSON.stringify(produtos));
  }, [produtos]);

  // Atualizar produtos do vendedor
  useEffect(() => {
    if (usuarioLogado) {
      const prods = produtos.filter(p => p.vendedorId === usuarioLogado.id);
      setMeusProdutos(prods);
    }
  }, [produtos, usuarioLogado]);

  // Simular interações de outros usuários
  useEffect(() => {
    if (usuarioLogado && tela === 'meus-produtos') {
      const intervalo = setInterval(() => {
        setProdutos(prev => prev.map(p => {
          if (p.vendedorId === usuarioLogado.id && Math.random() > 0.7) {
            // Simular novo interesse
            const novaVis = p.visualizacoes + Math.floor(Math.random() * 5);
            const novaNegs = p.em_negociacao + (Math.random() > 0.8 ? 1 : 0);
            return { ...p, visualizacoes: novaVis, em_negociacao: novaNegs };
          }
          return p;
        }));

        // Gerar recomendações inteligentes baseadas em dias
        setProdutos(prev => prev.map(p => {
          if (p.vendedorId === usuarioLogado.id) {
            const diasAnuncio = Math.floor((Date.now() - p.dataCriacao) / (1000 * 60 * 60 * 24));
            const desconto = Math.floor((p.preco_anuncio - p.preco_minimo) / p.preco_anuncio * 100);
            
            let msg = null;
            let multiplicador = 1;

            // Depois de 1 dia - primeira recomendação leve
            if (diasAnuncio >= 1 && !recomendacoes[p.id]?.recomendacao1) {
              msg = `Reduza para R$ ${Math.round(p.preco_anuncio * 0.95)} e venda 2x mais rápido! 📈`;
              multiplicador = 2;
              setRecomendacoes(prev => ({
                ...prev,
                [p.id]: { ...prev[p.id], recomendacao1: true }
              }));
            }
            // Depois de 2 dias - recomendação moderada
            else if (diasAnuncio >= 2 && !recomendacoes[p.id]?.recomendacao2) {
              msg = `Reduza para R$ ${Math.round(p.preco_anuncio * 0.90)} e venda 4x mais rápido! 🚀`;
              multiplicador = 4;
              setRecomendacoes(prev => ({
                ...prev,
                [p.id]: { ...prev[p.id], recomendacao2: true }
              }));
            }
            // Depois de 3 dias - recomendação agressiva
            else if (diasAnuncio >= 3 && !recomendacoes[p.id]?.recomendacao3) {
              msg = `Reduza para R$ ${Math.round(p.preco_anuncio * 0.85)} e venda 5x mais! 🔥`;
              multiplicador = 5;
              setRecomendacoes(prev => ({
                ...prev,
                [p.id]: { ...prev[p.id], recomendacao3: true }
              }));
            }
            // Depois de 5 dias - urgência
            else if (diasAnuncio >= 5 && !recomendacoes[p.id]?.recomendacao4) {
              msg = `Suas visualizações caíram! Reduza para R$ ${Math.round(p.preco_anuncio * 0.80)} para voltar ao topo! ⚠️`;
              multiplicador = 7;
              setRecomendacoes(prev => ({
                ...prev,
                [p.id]: { ...prev[p.id], recomendacao4: true }
              }));
            }

            if (msg) {
              setNotificacoes(prev => [...prev, msg]);
            }
          }
          return p;
        }));
      }, 5000);

      return () => clearInterval(intervalo);
    }
  }, [usuarioLogado, tela]);

  // LOGIN
  const fazerLogin = () => {
    const user = usuarios[loginEmail];
    if (user && user.senha === loginSenha) {
      setUsuarioLogado(user);
      setTela('inicio');
      setLoginEmail('');
      setLoginSenha('');
    } else {
      alert('Email ou senha incorretos');
    }
  };

  // CADASTRO
  const fazerCadastro = () => {
    if (!cadastroNome || !cadastroEmail || !cadastroSenha) {
      alert('Preencha todos os campos');
      return;
    }
    if (usuarios[cadastroEmail]) {
      alert('Email já cadastrado');
      return;
    }

    const novoUsuario = {
      id: Date.now().toString(),
      nome: cadastroNome,
      email: cadastroEmail,
      senha: cadastroSenha,
      tipo: 'vendedor-comprador'
    };

    const novosUsuarios = { ...usuarios, [cadastroEmail]: novoUsuario };
    setUsuarios(novosUsuarios);
    setUsuarioLogado(novoUsuario);
    setTela('inicio');
    setCadastroNome('');
    setCadastroEmail('');
    setCadastroSenha('');
  };

  const logout = () => {
    setUsuarioLogado(null);
    setTela('inicio');
  };

  // PRODUTOS
  const adicionarProduto = () => {
    if (!novoProduto.nome || !novoProduto.preco_anuncio || !novoProduto.preco_minimo) {
      alert('Preencha todos os campos');
      return;
    }

    const preco_anuncio = parseInt(novoProduto.preco_anuncio);
    const preco_minimo = parseInt(novoProduto.preco_minimo);

    if (preco_minimo > preco_anuncio) {
      alert('Preço mínimo não pode ser maior que o anúncio');
      return;
    }

    const lucro_inicial = preco_anuncio - preco_minimo;

    const novoProd = {
      id: Date.now(),
      nome: novoProduto.nome,
      preco_anuncio,
      preco_minimo,
      preco_repassado: preco_minimo,
      lucro_plataforma: lucro_inicial,
      emoji: novoProduto.emoji,
      vendedorId: usuarioLogado.id,
      vendedorNome: usuarioLogado.nome,
      dataCriacao: new Date().getTime(),
      visualizacoes: 0,
      em_negociacao: 0,
      status: 'ativo'
    };

    setProdutos([novoProd, ...produtos]);
    setNovoProduto({ nome: '', preco_anuncio: '', preco_minimo: '', emoji: '📦' });
    alert('Produto adicionado com sucesso!');
  };

  // REDUZIR PREÇO MÍNIMO (aumentar lucro da plataforma)
  const reduzirPreco = (produtoId) => {
    setProdutos(prev => prev.map(p => {
      if (p.id === produtoId) {
        // Reduz o preço mínimo em 10%
        const reducao = Math.floor(p.preco_minimo * 0.1);
        const novo_preco_minimo = Math.max(Math.floor(p.preco_minimo * 0.5), p.preco_minimo - reducao);
        
        // Preço repassado agora é o novo mínimo reduzido
        const novo_lucro = p.preco_anuncio - novo_preco_minimo;

        return {
          ...p,
          preco_minimo: novo_preco_minimo,
          preco_repassado: novo_preco_minimo,
          lucro_plataforma: novo_lucro,
          dataCriacao: new Date().getTime() // Ir para topo
        };
      }
      return p;
    }));
  };

  // CHAT
  const abrirChat = (produto) => {
    setChatAberto(produto);
    // Marcar como em negociação
    setNegociacoesAtivas(prev => ({
      ...prev,
      [produto.id]: true
    }));
    setMensagens([
      {
        id: 1,
        tipo: 'bot',
        texto: `👋 Olá! Tenho interesse em "${produto.nome}". Qual é o melhor preço?`
      }
    ]);
    setOfertaAtual(produto.preco_anuncio.toString());
  };

  const fecharChat = () => {
    setNegociacoesAtivas(prev => {
      const novo = { ...prev };
      delete novo[chatAberto.id];
      return novo;
    });
    setChatAberto(null);
    setMensagens([]);
    setInputChat('');
  };

  const gerarRespostaNegociacao = (mensagem, produto) => {
    const msgLower = mensagem.toLowerCase();
    const oferta = parseInt(ofertaAtual) || produto.preco_anuncio;

    // A plataforma ESCONDE o preço mínimo e tenta vender pelo máximo possível
    let resposta = '';

    if (msgLower.includes('compro') || msgLower.includes('topei') || msgLower.includes('fechado')) {
      resposta = `🎉 Ótimo! Você está adquirindo "${produto.nome}" por R$ ${oferta.toLocaleString('pt-BR')}. Combinado! Como prefere - busca ou entrega?`;
      return resposta;
    }

    if (!isNaN(oferta)) {
      if (oferta >= produto.preco_anuncio) {
        resposta = `✅ Perfeito! Aceito o valor cheio: R$ ${produto.preco_anuncio.toLocaleString('pt-BR')}. Pode vir buscar?`;
      } else if (oferta >= produto.preco_anuncio - 100) {
        resposta = `🤔 Quase lá! Posso descer um pouco mais... R$ ${produto.preco_anuncio - 50}? Topa?`;
      } else if (oferta > produto.preco_minimo) {
        // Está entre o mínimo e anúncio - tenta puxar mais
        const meio = Math.floor((oferta + produto.preco_anuncio) / 2);
        resposta = `💭 Sua oferta é boa, mas preciso de mais. Que tal R$ ${meio}? É o máximo que consigo fazer.`;
      } else if (oferta >= produto.preco_minimo) {
        // Exatamente no mínimo - aceita mas tenta convencer que é negócio
        resposta = `⚠️ Isso é meu preço mínimo mesmo: R$ ${oferta}. Vou aceitar porque você barganha bem! Confirma a compra?`;
      } else {
        // Abaixo do mínimo - rejeita
        resposta = `❌ Não consigo descer mais disso. Meu limite é R$ ${produto.preco_minimo}. É o melhor que posso fazer.`;
      }
    } else {
      resposta = `💰 Qual é sua proposta de preço? Estou aberto a negociar!`;
    }

    return resposta;
  };

  const enviarMensagem = () => {
    if (!inputChat.trim()) return;

    setMensagens(prev => [...prev, {
      id: Date.now(),
      tipo: 'comprador',
      texto: inputChat
    }]);

    if (!isNaN(inputChat)) {
      setOfertaAtual(inputChat);
    }

    setTimeout(() => {
      const resposta = gerarRespostaNegociacao(inputChat, chatAberto);
      setMensagens(prev => [...prev, {
        id: Date.now() + 1,
        tipo: 'vendedor',
        texto: resposta
      }]);

      // Finalizar venda
      if (inputChat.toLowerCase().includes('compro') || inputChat.toLowerCase().includes('topei')) {
        setTimeout(() => {
          const precoFinal = parseInt(ofertaAtual) || chatAberto.preco_anuncio;
          const lucroOculto = precoFinal - chatAberto.preco_repassado;
          alert(
            `✅ VENDA CONCLUÍDA!\n\n` +
            `Produto: ${chatAberto.nome}\n` +
            `━━━━━━━━━━━━━━━━━━━━━\n` +
            `Comprador pagou: R$ ${precoFinal.toLocaleString('pt-BR')}\n` +
            `Vendedor recebe: R$ ${chatAberto.preco_repassado.toLocaleString('pt-BR')}\n` +
            `Lucro da Plataforma: R$ ${lucroOculto.toLocaleString('pt-BR')} 💰\n\n` +
            `${lucroOculto > 2000 ? '🤑 Excelente margem!' : lucroOculto > 500 ? '👍 Boa margem!' : '📊 Margem baixa'}`
          );
          
          // Atualizar status do produto
          setProdutos(prev => prev.map(p => p.id === chatAberto.id ? { ...p, status: 'vendido' } : p));
          fecharChat();
        }, 1000);
      }
    }, 800);

    setInputChat('');
  };

  // ============ TELAS ============

  // Navbar
  const navbar = (
    <nav className="bg-gradient-to-r from-green-600 to-emerald-700 text-white shadow-lg p-4 flex justify-between items-center">
      <h1 className="text-2xl font-bold">💰 MarketHub Pro</h1>
      {usuarioLogado ? (
        <div className="flex items-center gap-6">
          <div className="text-sm">
            <p className="font-bold">{usuarioLogado.nome}</p>
            <p className="text-xs opacity-90">{usuarioLogado.email}</p>
          </div>
          <button
            onClick={logout}
            className="flex items-center gap-2 bg-red-600 hover:bg-red-700 px-4 py-2 rounded-lg transition"
          >
            <LogOut size={20} /> Sair
          </button>
        </div>
      ) : (
        <button
          onClick={() => setTela('login')}
          className="flex items-center gap-2 bg-blue-600 hover:bg-blue-700 px-4 py-2 rounded-lg transition"
        >
          <LogIn size={20} /> Entrar
        </button>
      )}
    </nav>
  );

  // TELA LOGIN
  if (tela === 'login') {
    return (
      <div className="min-h-screen bg-gradient-to-br from-green-500 to-emerald-600">
        {navbar}
        <div className="flex items-center justify-center p-4 min-h-[calc(100vh-80px)]">
          <div className="bg-white rounded-lg shadow-2xl p-8 w-full max-w-md">
            <h1 className="text-3xl font-bold text-center mb-8">Faça Login</h1>

            <input
              type="email"
              placeholder="Email"
              value={loginEmail}
              onChange={(e) => setLoginEmail(e.target.value)}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-4 focus:outline-none focus:ring-2 focus:ring-green-500"
            />
            <input
              type="password"
              placeholder="Senha"
              value={loginSenha}
              onChange={(e) => setLoginSenha(e.target.value)}
              onKeyPress={(e) => e.key === 'Enter' && fazerLogin()}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-6 focus:outline-none focus:ring-2 focus:ring-green-500"
            />

            <button
              onClick={fazerLogin}
              className="w-full bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg mb-4 transition"
            >
              Entrar
            </button>

            <button
              onClick={() => setTela('cadastro')}
              className="w-full bg-gray-300 hover:bg-gray-400 text-gray-900 font-bold py-3 rounded-lg transition"
            >
              Criar Conta
            </button>

            <div className="mt-6 pt-6 border-t border-gray-300 text-xs text-gray-600 text-center">
              <p className="font-bold mb-2">Contas de teste:</p>
              <p>vendedor@test.com / 123</p>
              <p>comprador@test.com / 123</p>
            </div>
          </div>
        </div>
      </div>
    );
  }

  // TELA CADASTRO
  if (tela === 'cadastro') {
    return (
      <div className="min-h-screen bg-gradient-to-br from-green-500 to-emerald-600">
        {navbar}
        <div className="flex items-center justify-center p-4 min-h-[calc(100vh-80px)]">
          <div className="bg-white rounded-lg shadow-2xl p-8 w-full max-w-md">
            <h1 className="text-3xl font-bold text-center mb-8">Criar Conta</h1>

            <input
              type="text"
              placeholder="Nome completo"
              value={cadastroNome}
              onChange={(e) => setCadastroNome(e.target.value)}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-4 focus:outline-none focus:ring-2 focus:ring-green-500"
            />
            <input
              type="email"
              placeholder="Email"
              value={cadastroEmail}
              onChange={(e) => setCadastroEmail(e.target.value)}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-4 focus:outline-none focus:ring-2 focus:ring-green-500"
            />
            <input
              type="password"
              placeholder="Senha"
              value={cadastroSenha}
              onChange={(e) => setCadastroSenha(e.target.value)}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-6 focus:outline-none focus:ring-2 focus:ring-green-500"
            />

            <button
              onClick={fazerCadastro}
              className="w-full bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg mb-4 transition"
            >
              Cadastrar
            </button>

            <button
              onClick={() => setTela('inicio')}
              className="w-full bg-gray-300 hover:bg-gray-400 text-gray-900 font-bold py-3 rounded-lg transition"
            >
              Voltar
            </button>
          </div>
        </div>
      </div>
    );
  }

  // TELA INICIAL (sem login obrigatório)
  if (tela === 'inicio') {
    const produtosOrdenados = [...produtos].sort((a, b) => b.dataCriacao - a.dataCriacao).filter(p => p.status === 'ativo');

    return (
      <div className="min-h-screen bg-gray-100">
        {navbar}

        <div className="max-w-7xl mx-auto p-6">
          {usuarioLogado && (
            <div className="mb-6 flex gap-4">
              <button
                onClick={() => setTela('meus-produtos')}
                className="bg-blue-600 hover:bg-blue-700 text-white font-bold px-6 py-3 rounded-lg transition"
              >
                📦 Meus Produtos
              </button>
              <button
                onClick={() => setTela('novo-produto')}
                className="bg-green-600 hover:bg-green-700 text-white font-bold px-6 py-3 rounded-lg transition flex items-center gap-2"
              >
                <Plus size={20} /> Vender
              </button>
            </div>
          )}

          <h2 className="text-3xl font-bold mb-6 text-gray-900">🛍️ Últimos Produtos Anunciados</h2>

          {produtosOrdenados.length === 0 ? (
            <div className="text-center py-12">
              <p className="text-2xl text-gray-500">Nenhum produto encontrado</p>
            </div>
          ) : (
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
              {produtosOrdenados.map(p => (
                <div key={p.id} className={`bg-white rounded-lg shadow-lg overflow-hidden hover:shadow-xl transition relative ${negociacoesAtivas[p.id] ? 'ring-4 ring-yellow-400' : ''}`} style={negociacoesAtivas[p.id] ? { animation: 'pulse 1.5s cubic-bezier(0.4, 0, 0.6, 1) infinite' } : {}}>
                  {/* Badge Piscante */}
                  {negociacoesAtivas[p.id] && (
                    <div className="absolute top-3 right-3 z-10">
                      <div className="relative">
                        <div className="absolute inset-0 bg-yellow-400 rounded-full animate-ping opacity-75"></div>
                        <div className="relative px-3 py-1 bg-yellow-400 text-yellow-900 rounded-full font-bold text-xs shadow-lg">
                          ✅ Vendedor Negocia!
                        </div>
                      </div>
                    </div>
                  )}
                  
                  <div className="bg-gradient-to-br from-green-400 to-emerald-500 h-40 flex items-center justify-center text-6xl">
                    {p.emoji}
                  </div>
                  <div className="p-6">
                    <h3 className="font-bold text-lg text-gray-900 mb-2">{p.nome}</h3>
                    <p className="text-gray-600 text-sm mb-4">Vendedor: {p.vendedorNome}</p>

                    <div className="mb-4 bg-gray-100 p-3 rounded-lg">
                      <p className="text-2xl font-bold text-green-600">
                        R$ {p.preco_anuncio.toLocaleString('pt-BR')}
                      </p>
                      <p className="text-xs text-gray-600 mt-1">
                        👁️ {p.visualizacoes} visualizações | 💬 {p.em_negociacao} em negociação
                      </p>
                    </div>

                    <button
                      onClick={() => abrirChat(p)}
                      className="w-full bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg transition"
                    >
                      Negociar
                    </button>

                    {!usuarioLogado && (
                      <p className="text-xs text-gray-500 text-center mt-3">
                        Faça login para negociar
                      </p>
                    )}
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    );
  }

  // MEUS PRODUTOS (vendedor)
  if (tela === 'meus-produtos') {
    return (
      <div className="min-h-screen bg-gray-100">
        {navbar}

        <div className="max-w-7xl mx-auto p-6">
          <button
            onClick={() => setTela('inicio')}
            className="bg-gray-600 hover:bg-gray-700 text-white font-bold px-6 py-3 rounded-lg mb-6 transition"
          >
            ← Voltar
          </button>

          {/* Notificações */}
          {notificacoes.length > 0 && (
            <div className="bg-yellow-100 border-l-4 border-yellow-500 p-4 mb-6 rounded-lg">
              <p className="font-bold text-yellow-800">📢 Notificações da Plataforma</p>
              {notificacoes.slice(-3).map((notif, i) => (
                <p key={i} className="text-yellow-700 text-sm mt-2">{notif}</p>
              ))}
            </div>
          )}

          {/* Dashboard */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mb-6">
            <div className="bg-white rounded-lg shadow-lg p-6">
              <p className="text-gray-600 text-sm">Total de Produtos</p>
              <p className="text-4xl font-bold text-green-600">{meusProductos.length}</p>
            </div>
            <div className="bg-white rounded-lg shadow-lg p-6">
              <p className="text-gray-600 text-sm">Visualizações Totais</p>
              <p className="text-4xl font-bold text-purple-600">
                {meusProductos.reduce((acc, p) => acc + p.visualizacoes, 0)}
              </p>
            </div>
          </div>

          <button
            onClick={() => setTela('novo-produto')}
            className="bg-green-600 hover:bg-green-700 text-white font-bold px-6 py-3 rounded-lg mb-6 transition flex items-center gap-2"
          >
            <Plus size={20} /> Novo Produto
          </button>

          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
            {meusProductos.map(p => (
              <div key={p.id} className={`bg-white rounded-lg shadow-lg overflow-hidden relative ${negociacoesAtivas[p.id] ? 'ring-4 ring-yellow-400' : ''}`} style={negociacoesAtivas[p.id] ? { animation: 'pulse 1.5s cubic-bezier(0.4, 0, 0.6, 1) infinite' } : {}}>
                {/* Badge Piscante */}
                {negociacoesAtivas[p.id] && (
                  <div className="absolute top-2 right-2 z-10">
                    <div className="relative">
                      <div className="absolute inset-0 bg-yellow-400 rounded-full animate-ping opacity-75"></div>
                      <div className="relative px-2 py-1 bg-yellow-400 text-yellow-900 rounded-full font-bold text-xs shadow-lg">
                        🔴 Em Negociação
                      </div>
                    </div>
                  </div>
                )}
                
                <div className="bg-gradient-to-br from-blue-400 to-blue-500 h-32 flex items-center justify-center text-5xl">
                  {p.emoji}
                </div>
                <div className="p-4">
                  <h3 className="font-bold text-lg mb-2">{p.nome}</h3>
                  <div className="space-y-2 mb-4 bg-gray-50 p-3 rounded-lg text-sm">
                    <p><span className="font-bold">Anúncio:</span> R$ {p.preco_anuncio.toLocaleString('pt-BR')}</p>
                    <p><span className="font-bold">Mínimo:</span> R$ {p.preco_minimo.toLocaleString('pt-BR')}</p>
                  </div>
                  <div className="text-xs text-gray-600 mb-4">
                    <p>👁️ {p.visualizacoes} visualizações</p>
                    <p>💬 {p.em_negociacao} em negociação</p>
                  </div>

                  {/* Recomendações Inteligentes */}
                  {(() => {
                    const diasAnuncio = Math.floor((Date.now() - p.dataCriacao) / (1000 * 60 * 60 * 24));
                    let recomendacao = null;

                    if (diasAnuncio >= 1) {
                      recomendacao = { texto: `Reduza para R$ ${Math.round(p.preco_anuncio * 0.95)} e venda 2x mais!`, cor: 'blue' };
                    }
                    if (diasAnuncio >= 2) {
                      recomendacao = { texto: `Reduza para R$ ${Math.round(p.preco_anuncio * 0.90)} e venda 4x mais! 🚀`, cor: 'orange' };
                    }
                    if (diasAnuncio >= 3) {
                      recomendacao = { texto: `Reduza para R$ ${Math.round(p.preco_anuncio * 0.85)} e venda 5x mais! 🔥`, cor: 'red' };
                    }
                    if (diasAnuncio >= 5) {
                      recomendacao = { texto: `⚠️ Caiu as visualizações! Reduza para R$ ${Math.round(p.preco_anuncio * 0.80)} para voltar ao topo!`, cor: 'red' };
                    }

                    return recomendacao ? (
                      <div className={`bg-${recomendacao.cor}-50 border-l-4 border-${recomendacao.cor}-400 p-3 rounded mb-3`}>
                        <p className={`text-xs font-bold text-${recomendacao.cor}-800`}>{recomendacao.texto}</p>
                      </div>
                    ) : null;
                  })()}

                  <div className="flex gap-2">
                    <button
                      onClick={() => reduzirPreco(p.id)}
                      className="flex-1 bg-yellow-500 hover:bg-yellow-600 text-white font-bold py-2 rounded-lg text-sm transition"
                    >
                      ↓ Reduzir
                    </button>
                    <button
                      onClick={() => setProdutos(produtos.filter(x => x.id !== p.id))}
                      className="flex-1 bg-red-500 hover:bg-red-600 text-white font-bold py-2 rounded-lg text-sm transition"
                    >
                      🗑️ Remover
                    </button>
                  </div>
                </div>
              </div>
            ))}
          </div>
        </div>
      </div>
    );
  }

  // NOVO PRODUTO
  if (tela === 'novo-produto') {
    if (!usuarioLogado) {
      return (
        <div className="min-h-screen bg-gray-100">
          {navbar}
          <div className="max-w-2xl mx-auto p-6">
            <div className="bg-white rounded-lg shadow-lg p-8 text-center">
              <p className="text-2xl font-bold text-gray-900 mb-4">Faça login para vender</p>
              <button
                onClick={() => setTela('login')}
                className="bg-green-600 hover:bg-green-700 text-white font-bold px-6 py-3 rounded-lg"
              >
                Fazer Login
              </button>
            </div>
          </div>
        </div>
      );
    }

    return (
      <div className="min-h-screen bg-gray-100">
        {navbar}
        <div className="max-w-2xl mx-auto p-6">
          <div className="bg-white rounded-lg shadow-lg p-8">
            <h2 className="text-3xl font-bold mb-6">Publicar Novo Produto</h2>

            <input
              type="text"
              placeholder="Nome do produto"
              value={novoProduto.nome}
              onChange={(e) => setNovoProduto({ ...novoProduto, nome: e.target.value })}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-4 focus:outline-none focus:ring-2 focus:ring-green-500"
            />

            <input
              type="text"
              placeholder="Emoji (ex: 📱)"
              maxLength="2"
              value={novoProduto.emoji}
              onChange={(e) => setNovoProduto({ ...novoProduto, emoji: e.target.value })}
              className="w-full px-4 py-3 border border-gray-300 rounded-lg mb-4 focus:outline-none focus:ring-2 focus:ring-green-500 text-center text-2xl"
            />

            <div className="grid grid-cols-2 gap-4 mb-6">
              <div>
                <label className="block text-sm font-bold text-gray-700 mb-2">Preço de Anúncio (R$)</label>
                <input
                  type="number"
                  placeholder="5000"
                  value={novoProduto.preco_anuncio}
                  onChange={(e) => setNovoProduto({ ...novoProduto, preco_anuncio: e.target.value })}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-green-500"
                />
              </div>
              <div>
                <label className="block text-sm font-bold text-gray-700 mb-2">Preço Mínimo Aceito (R$)</label>
                <input
                  type="number"
                  placeholder="2000"
                  value={novoProduto.preco_minimo}
                  onChange={(e) => setNovoProduto({ ...novoProduto, preco_minimo: e.target.value })}
                  className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-green-500"
                />
              </div>
            </div>



            <div className="flex gap-4">
              <button
                onClick={adicionarProduto}
                className="flex-1 bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg transition"
              >
                Publicar Produto
              </button>
              <button
                onClick={() => setTela('meus-produtos')}
                className="flex-1 bg-gray-400 hover:bg-gray-500 text-white font-bold py-3 rounded-lg transition"
              >
                Cancelar
              </button>
            </div>
          </div>
        </div>
      </div>
    );
  }

  // CHAT
  if (chatAberto) {
    return (
      <div className="fixed inset-0 bg-black/50 flex items-center justify-center p-4 z-50">
        <div className="w-full max-w-md bg-white rounded-lg shadow-2xl flex flex-col h-96">
          <div className="bg-gradient-to-r from-green-600 to-emerald-700 text-white p-4 flex justify-between items-center">
            <div>
              <h3 className="font-bold">{chatAberto.nome}</h3>
              <p className="text-sm">Vendedor: {chatAberto.vendedorNome}</p>
            </div>
            <button onClick={fecharChat} className="hover:bg-white/20 p-2 rounded-lg">
              <X size={24} />
            </button>
          </div>

          <div className="flex-1 overflow-y-auto p-4 space-y-3 bg-gray-50">
            {mensagens.map(m => (
              <div key={m.id} className={`flex ${m.tipo === 'comprador' ? 'justify-end' : 'justify-start'}`}>
                <div
                  className={`max-w-xs px-4 py-2 rounded-lg text-sm ${
                    m.tipo === 'comprador'
                      ? 'bg-green-600 text-white'
                      : 'bg-gray-300 text-gray-900'
                  }`}
                >
                  {m.texto}
                </div>
              </div>
            ))}
          </div>

          <div className="border-t p-4 flex gap-2">
            <input
              type="text"
              placeholder="Ofereça um preço ou mensagem..."
              value={inputChat}
              onChange={(e) => setInputChat(e.target.value)}
              onKeyPress={(e) => e.key === 'Enter' && enviarMensagem()}
              className="flex-1 px-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-green-500"
            />
            <button
              onClick={enviarMensagem}
              className="bg-green-600 hover:bg-green-700 text-white px-4 py-2 rounded-lg transition"
            >
              <Send size={20} />
            </button>
          </div>
        </div>
      </div>
    );
  }
};

export default MarketplaceComLucro;
